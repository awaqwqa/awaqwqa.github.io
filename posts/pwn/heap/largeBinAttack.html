static void

{
  INTERNAL_SIZE_T size;        /* its size */
  mfastbinptr *fb;             /* associated fastbin */
  mchunkptr nextchunk;         /* next contiguous chunk */
  INTERNAL_SIZE_T nextsize;    /* its size */
  int nextinuse;               /* true if nextchunk is used */
  INTERNAL_SIZE_T prevsize;    /* size of previous contiguous chunk */
  mchunkptr bck;               /* misc temp for linking */

  size = chunksize (p);

     allocator never wrapps around at the end of the address space.
     Therefore we can exclude some size values which might appear
     here by accident or by "design" from some intruder.  */
  if (__builtin_expect ((uintptr_t) p > (uintptr_t) -size, 0)
      || __builtin_expect (misaligned_chunk (p), 0))
    malloc_printerr ("free(): invalid pointer");
  
  /* We know that each chunk is at least MINSIZE bytes in size or a
     multiple of MALLOC_ALIGNMENT.  */

  if (__glibc_unlikely (size < MINSIZE || !aligned_OK (size)))
    malloc_printerr ("free(): invalid size");

  check_inuse_chunk(av, p);

#if USE_TCACHE
  {
    size_t tc_idx = csize2tidx (size);

    if (tcache
	&& tc_idx < mp_.tcache_bins
	&& tcache->counts[tc_idx] < mp_.tcache_count)
      {
	tcache_put (p, tc_idx);
	return;
      }
  }
#endif

  /*
    If eligible, place chunk on a fastbin so it can be found
    and used quickly in malloc.
  */
    //  å¦‚æœæ˜¯fastbinåŒºé—´çš„
  if ((unsigned long)(size) <= (unsigned long)(get_max_fast ())

#if TRIM_FASTBINS
      /*
	If TRIM_FASTBINS set, don't place chunks
	bordering top into fastbins
      */
      && (chunk_at_offset(p, size) != av->top)
#endif
      ) {
    // chunkçš„sizeå€¼ å¾—å¤§äºchunkæœ€å°å€¼ å¾—å°äºè¯¥åŒºåŸŸåˆ†é…çš„æœ€å¤§size

	    bool fail = true;

	   of system_mem might result in a false positive.  Redo the test after
	   getting the lock.  */
        if (!have_lock){
            __libc_lock_lock (av->mutex);
            fail = (chunksize_nomask (chunk_at_offset (p, size)) <= 2 * SIZE_SZ
                || chunksize (chunk_at_offset (p, size)) >= av->system_mem);
            __libc_lock_unlock (av->mutex);
        }

        if (fail)
            malloc_printerr ("free(): invalid next size (fast)");
    }

    mchunkptr old = *fb, old2;

    if (SINGLE_THREAD_P){

	   add (i.e., double free).  */
       // æ£€æŸ¥fastbinçš„å¤´éƒ¨chunkæ˜¯å¦ä¸ºå½“å‰freeçš„chunk

            malloc_printerr ("double free or corruption (fasttop)");
        // p->old
        p->fd = old;
        *fb = p;

        /* Check that the top of the bin is not the record we are going to
            add (i.e., double free).  */

                malloc_printerr ("double free or corruption (fasttop)");
            p->fd = old2 = old;
        }while ((old = catomic_compare_and_exchange_val_rel (fb, p, old2))
            != old2);

        size of the chunk that we are adding.  We can dereference OLD
        only if we have the lock, otherwise it might have already been
        allocated again.  */
        if (have_lock && old != NULL&& __builtin_expect (fastbin_index (chunksize (old)) != idx, 0))
            malloc_printerr ("invalid fastbin entry (free)");

  /*

  */
    // å¦‚æœé‡Šæ”¾çš„chunkä¸å±äºfastbin ä¸”ä¸æ˜¯mmapåˆ†é…çš„ å°±è·å–ä¸‹ä¸€ä¸ªchunkçš„æŒ‡é’ˆ nextchunkå’Œnextsize
    // å¦‚æœå‰ä¸€ä¸ªchunkç©ºé—² å°±åˆå¹¶ é€šè¿‡unlinkå°†è¯¥chunkè„±ç¦»å‡ºæ¥
    // å¦‚æœå–å‡ºæ¥çš„chunkä¸‹ä¸€ä¸ªchunkä¹Ÿæ˜¯free chunk ä¸”ä¸ä¸ºtop chunk åˆ™ä¹Ÿè®¾ç½®ä¸ºç©ºé—²

    /* If we're single-threaded, don't lock the arena.  */
    if (SINGLE_THREAD_P)
      have_lock = true;

    if (!have_lock)
      __libc_lock_lock (av->mutex);

       top block.  */

    // æŸ¥çœ‹ä¸‹ä¸€ä¸ªchunkæ˜¯å¦å¤§äºæ•´ä¸ªå†…å­˜ç©ºé—´çš„è¾¹ç•Œ
    if (__builtin_expect (contiguous (av)
			  && (char *) nextchunk
			  >= ((char *) av->top + chunksize(av->top)), 0))

	    (unsigned long)(mp_.trim_threshold))

	   large, because the corresponding heap might go away.  */

      __libc_lock_unlock (av->mutex);
  }
  /*

  */

  else {
    munmap_chunk (p);
  }
}
